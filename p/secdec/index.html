uint32 hamming_distance( unint32 a, unit32 b) {
    uint32 num = 0;
    uint32 diff;
    diff = a ^ b;
    while (diff) {
		num ++;
		diff = diff & (diff -1 ); //remove the leading 1
	}
	return num;
}
